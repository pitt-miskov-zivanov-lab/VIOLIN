[docs]def match_level(x,reading_df,model_df):
    """
    This function calculates the Match Level for an interaction from the reading

    Parameters
    ----------
    x : int
        The line of the reading dataframe with the interaction to be scored
    reading_df : pd.DataFrame
        The reading dataframe
    model_df : pd.DataFrame
        The model dataframe

    Returns
    -------
    match : int
        Match Level score value
    """

    #presence of regulated/regulator elements defaults to False
    regulated = False
    regulator = False
    if reading_df.loc[x,'PosReg Name']=="nan": reg_name = 'NegReg'
    else: reg_name = 'PosReg'

    #Search for regulated from reading in model
    if (find_element_index("name",reading_df.loc[x,'Element Name'],reading_df.loc[x,'Element Type'],model_df) != -1 or 
        find_element_index("id",reading_df.loc[x,'Element Identifier'],reading_df.loc[x,'Element Type'],model_df) != -1):
        regulated = True

    #Search for regulator from reading in model
    if (find_element_index("name",reading_df.loc[x,reg_name+' Name'],reading_df.loc[x,reg_name+' Type'],model_df) != -1 or 
        find_element_index("id",reading_df.loc[x,reg_name+' ID'],reading_df.loc[x,reg_name+' Type'],model_df) != -1):
        regulator = True

    #Scoring Definition
    #regulator and regulated in model
    if regulated and regulator: match = 10
    #regulator not in the model
    elif regulated and not regulator: match = 100
    #regulated not in the model
    elif not regulated and regulator: match = 1
    #neither regulated nor regulator in model
    else: match = 0.1

    return match




[docs]def kind_score(x,model_df,reading_df,graph):
    """
    This function calculates the Kind Score for an interaction in the reading

    Parameters
    ----------
    x : int
        The line of the reading dataframe with the interaction to be scored
    model_df: pd.DataFrame
        The model dataframe
    reading_df : pd.DataFrame
        The reading dataframe
    graph : nx.Graph
        edge list of the model,used when function calls path_finding module

    Returns
    -------
    kind : int
        Kind Score score value
    """
    #Determines whether regulator is positive or negative
    if reading_df.at[x,'PosReg Name'] == "nan": 
        regulator = ['NegReg Name','NegReg Type','NegReg ID',
                    'NegReg','NegReg IDs','PosReg','PosReg IDs']
    else: regulator = ['PosReg Name','PosReg Type','PosReg ID',
                    'PosReg','PosReg IDs','NegReg','NegReg IDs']

    #List of Regulated and Regulator Name, Type, ID
    nodes = [reading_df.at[x,'Element Name'],reading_df.at[x,'Element Type'],
            reading_df.at[x,'Element Identifier'],reading_df.at[x,regulator[0]],
            reading_df.at[x,regulator[1]],reading_df.at[x,regulator[2]]]
    #List of interaction attributes (location, location ID, cell line, cell type, organism)
    reading_atts = [reading_df.at[x,'Location'],reading_df.at[x,'Location Identifier'],
            reading_df.at[x,'Cell Line'],reading_df.at[x,'Cell Type'],reading_df.at[x,'Organism']]

    #Both regulator and regulated node found in the model
    if ((find_element_index("name",nodes[0],nodes[1],model_df)!=-1 or 
        find_element_index("id",nodes[2],nodes[1],model_df)!=-1) and 
        (find_element_index("name",nodes[3],nodes[4],model_df)!=-1 or 
        find_element_index("id",nodes[5],nodes[4],model_df)!=-1)):

       #Determine if regulated is found as name or ID 
        if (find_element_index("name",nodes[0],nodes[1],model_df)!=-1): 
            idx = find_element_index("name",nodes[0],nodes[1],model_df)
        else: idx = find_element_index("id",nodes[2],nodes[1],model_df)

        #Determine if regulator from reading is also a regulator in the model
        if {nodes[3] in str(model_df.loc[idx,regulator[3]]) or 
            nodes[5] in str(model_df.loc[idx,regulator[4]])}:

            #Generate list of model attributes to compare to reading attributes
            model_atts = [model_df.at[idx,'Element Type'], model_df.at[idx,'Location'],
                        model_df.at[idx,'Location ID'],model_df.at[idx,'Cell line'],
                        model_df.at[idx,'Cell type'],model_df.at[idx,'Organism']] 
            compare_atts = []

            #Create n-tuple of the results of compare() for each attribute 
            for x in range(len(reading_atts)): compare_atts.append(compare(model_atts[x],reading_atts[x]))
            #Strong Corroboration - perfect match
            if 0 in compare_atts and len(set(compare_atts)) == 1: kind = 2
            #Specification - reading contains information the model does not
            elif 1 in compare_atts and len(set(compare_atts)) == 1: kind = 30
            #Weak Corroboration - model contains more information than the reading
            elif 2 in compare_atts and len(set(compare_atts)) == 1: kind = 1
            #Contradiction or questionable - some or all of the attributes differ
            elif 3 in compare_atts and len(set(compare_atts)) == 1: kind = 10
            #Weak Corroboration - attributes either match or the model contains more information
            elif 0 in compare_atts and 2 in compare_atts and len(set(compare_atts)) == 2: kind = 1
            #Specification - attributes either match or the reading contains more information
            elif 0 in compare_atts and 1 in compare_atts and len(set(compare_atts)) == 2: kind = 30
            #Specification - model or reading contains more information, depending on attribute
            elif 2 in compare_atts and 1 in compare_atts and len(set(compare_atts)) == 2: kind = 30
            #Specification - combination of perfect match, model contains more information, 
            # reading contains more information
            else: kind = 30

        #Regulator is in the wrong direction
        elif {nodes[3] in str(model_df.loc[idx,regulator[5]]) or 
              nodes[5] in str(model_df.loc[idx,regulator[6]])}: 
            #If it's an indirect interaction, it's possible that direct regulation has one action, 
            # but indirect has another.
            if reading_df.at[x,'Interaction  Direct (D) or Indirect (I)'] == "i":
                kind = path_finding(nodes[3],regulator[0],nodes[4],nodes[5],nodes[0],nodes[1],nodes[2],model_df,graph)
            #Contradiction - If direct, then definitely a contradiction.
            else:kind = 10
    
        #Regulator from readin not in regulator lists in the model - look for path 
        else: 
            #If path is found, score determined by path_finding()
            if reading_df.at[x,'Interaction  Direct (D) or Indirect (I)'] == "i":
                kind = path_finding(nodes[3],regulator[0],nodes[4],nodes[5],nodes[0],nodes[1],nodes[2],model_df,graph)
            #Extension - no path found
            else:kind = 40

    #Extension - One or both nodes from reading not in model
    else: kind = 40

    return kind

[docs]def score_reading(reading_df,model_df,graph):
    """
    Creates new columns for the Match Level, Kind Score, Epistemic Value, and Total Score.
    Calls scoring functions and stores the values in the approriate column.

    Parameters
    ----------
    reading_df : pd.DataFrame
        The reading dataframe
    model_df : pd.DataFrame
        The model dataframe
    graph : nx.Graph
        edge list of the model, necessary for calling kind_score module

    Returns
    -------
    reading_df : pd.DataFrame
        reading dataframe with added scores
    """
    #Create new DF columns for score calculations
    reading_df['Match Level'] = pd.Series()
    reading_df['Kind Score'] = pd.Series()
    reading_df['Epistemic Value'] = pd.Series()
    reading_df['Total Score'] = pd.Series()

    #Calculate scores
    for x in range(reading_df.shape[0]):
        reading_df.at[x,'Match Level'] = match_level(x,reading_df,model_df)
        reading_df.at[x,'Kind Score'] = kind_score(x,model_df,reading_df,graph)
        reading_df.at[x,'Epistemic Value'] = 1
        reading_df.at[x,'Total Score'] =  {((reading_df.at[x,'Evidence Count']*
                                            reading_df.at[x,'Match Level'])+
                                            reading_df.at[x,'Kind Score'])*
                                            reading_df.at[x,'Epistemic Value']}

    return reading_df
