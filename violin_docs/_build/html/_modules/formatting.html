[docs]def evidence_count(reading_df, col_names):
    """
    This function merges duplicate interactions and finds evidence count in reading

    Parameters
    ----------
    reading_df : pd.DataFrame
        The dataframe of the machine reading output
    col_names: list
        The column headings used to determine if interactions are identical

    Returns
    -------
    counted_reading : pd.DataFrame
        A new dataframe with the evidence count and PMCID list for each interaction
    """
    #assemble the "interaction part" into a single column
    reading_df['interaction'] = reading_df[col_names].apply(lambda row: '_'.join(row.values.astype(str)), axis=1)
    reading_df['interaction'] = reading_df['interaction'].str.lower().str.replace('-','')
    
    #Compares 'interaction' column, and removes duplicates while merging PMC IDs into one cell
    counted_reading = reading_df.groupby('interaction')['Paper ID'].apply(np.array).reset_index(name='Paper ID')
    #Compares 'interaction' column, and removes duplicated while merging Evidence Text
    counted_evidence = reading_df.groupby('interaction')['Evidence'].apply(np.array).reset_index(name='Evidence')
    
    #Add Evidence to Dataframe 'r'
    counted_reading['Evidence'] = counted_evidence['Evidence']
    #create column for the number of times an interaction was found
    counted_reading['Evidence Count'] = counted_reading["Paper ID"].str.len()
    
    # split 'interaction' back into its appropriate columns
    interaction = counted_reading['interaction'].str.split("_", n = 21, expand = True)
    for each in col_names:
        counted_reading.insert(1+col_names.index(each), each, interaction[col_names.index(each)])
    return counted_reading




[docs]def add_regulator_names_id(model_df):
    """
    This function converts the regulator lists from BioRECIPE variables to element names and IDs

    Parameters
    ----------
    model_df : pd.DataFrame
        The model dataframe (in BioRECIPE format)

    Returns
    -------
    model_df : pd.DataFrame
        A new dataframe with added columns containing the positive and negative regulators as
        Element Names and IDs
    """
    #removes the initial values from the model dataframe, as they're not needed
    #Also adds new columns for the positive and negative regulator names and IDs
    col_headers = list(model_df.columns)
    end = col_headers.index('Negative Regulators')+1
    model_df = model_df.iloc[:,:end]
    model_df['PosReg'] = pd.Series().astype(object)
    model_df['PosReg IDs'] = pd.Series().astype(object)
    model_df['NegReg'] = pd.Series().astype(object)
    model_df['NegReg IDs'] = pd.Series().astype(object)

    #convert positive regulators
    for y in range(model_df.shape[0]):
        if model_df["Positive Regulators"][y]=="nan":
            model_df.at[y,'PosReg'] = "nan"
            model_df.at[y,'PosReg IDs'] = "nan"
        else:
            pos_reg_name = model_df["Positive Regulators"][y].split(',')
            pos_reg_id = []

            #formatting to remove special characters
            for z in range(len(pos_reg_name)):
                pos_reg_name[z] = pos_reg_name[z].replace("(","")
                pos_reg_name[z] = pos_reg_name[z].replace(")","")
                pos_reg_name[z] = pos_reg_name[z].replace("]","")
                pos_reg_name[z] = pos_reg_name[z].replace("{","")          
                if "}[" in pos_reg_name[z]:
                    regs = pos_reg_name[z].split("}[")
                    pos_reg_name.remove(pos_reg_name[z])
                    pos_reg_name.append(regs[0])
                    pos_reg_name.append(regs[1])
            pos_reg_var = pos_reg_name.copy()
            model_df.at[y,"Positive Regulators"] = pos_reg_var

            #find index for regulator in variable column, and copy the Element Name and IDs to the new columns
            for element in pos_reg_name:
                idx = list(model_df["Variable"]).index(element)
                pos_reg_name[pos_reg_name.index(element)] = model_df["Element Name"][idx]
                pos_reg_id.extend(model_df["Element IDs"][idx].split(","))
            
            model_df.at[y,'PosReg'] = pos_reg_name
            model_df.at[y,'PosReg IDs'] = pos_reg_id

    #convert negative regulators
    for y in range(model_df.shape[0]):
        if model_df["Negative Regulators"][y]=="nan":
            model_df.at[y,'NegReg'] = "nan"
            model_df.at[y,'NegReg IDs'] = "nan"
        else:
            neg_reg_name = model_df["Negative Regulators"][y].split(',')
            neg_reg_id = []

            #formatting to remove special characters
            for z in range(len(neg_reg_name)):
                neg_reg_name[z] = neg_reg_name[z].replace("(","")
                neg_reg_name[z] = neg_reg_name[z].replace(")","")
                neg_reg_name[z] = neg_reg_name[z].replace("]","")
                neg_reg_name[z] = neg_reg_name[z].replace("{","")          
                if "}[" in neg_reg_name[z]:###NOTE: Just remember that this notation is for complexes(?)
                    regs = neg_reg_name[z].split("}[")
                    neg_reg_name.remove(neg_reg_name[z])
                    neg_reg_name.append(regs[0])
                    neg_reg_name.append(regs[1])
            neg_reg_vars = neg_reg_name.copy()
            model_df.at[y,'Negative Regulators'] = neg_reg_vars

            #find index for regulator in variable column, and copy the Element Name and IDs to the new columns
            for element in neg_reg_name:
                idx = list(model_df["Variable"]).index(element)
                neg_reg_name[neg_reg_name.index(element)] = model_df["Element Name"][idx]
                neg_reg_id.extend(model_df["Element IDs"][idx].split(","))
            model_df.at[y,'NegReg'] = neg_reg_name
            model_df.at[y,'NegReg IDs'] = neg_reg_id
    return model_df
