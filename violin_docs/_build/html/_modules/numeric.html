[docs]def find_element_index(name_type, element_name, element_type,model_df):
    """
    This function finds the correct index of an element within the model.
    Because elements can exists as multiple types (Protein, RNA, gene, etc.),
    this function checks the element name/ID along with the element type

    Parameters
    ----------
    name_type : str
        Whether the element is being searched for by name or ID
    element_name: str
        The name (or ID) of the element being searched for
    element_type: str
        The type of element searched for (protein, protein family, etc.)
    model_df: pd.DataFrame
        The model dataframe

    Returns
    -------
    location : int
        The row of the model spreadsheet in which the element is found (returns -1 if not found)
    """
    location = -1
    if name_type == "name":
        indices = [i for i, x in enumerate(list(model_df['Element Name'])) if element_name in x]
        for idx in indices:
            if model_df.loc[idx,"Element Type"] == element_type or model_df.loc[idx,"Element Type"] in element_type or element_type in model_df.loc[idx,"Element Type"]: location = idx
    else:
        indices = [i for i, x in enumerate(list(model_df['Element IDs'])) if element_name in x]
        for idx in indices:
            if model_df.loc[idx,"Element Type"] == element_type or model_df.loc[idx,"Element Type"] in element_type or element_type in model_df.loc[idx,"Element Type"]: location = idx
    return location





[docs]def compare(model_att,reading_att):
    #TODO: need to be able to convert between GO IDs and location strings
    """
    Compares a model attribute to a reading attribute, returns numeric value\n
        Attributes are the same: 0 \n
        Model attribute missing, reading attribute present: 1\n
        Model attribute present, reading attribute missing: 2\n
        Model attribue differs from reading attribute: 3

    Parameters
    ----------
    model_att : str
        The model attribute
    reading_att : str
        The reading attribute

    Returns
    -------
    value : int
        Numerical representation of comparison outcome
    """

    if model_att == reading_att: value =  0
    elif (model_att == "nan") and (reading_att == "nan"): value = 0
    elif (model_att == "nan") and (reading_att != "nan"): value =  1
    elif (model_att != "nan") and (reading_att == "nan"): value =  2
    else: value =  3

    return value
