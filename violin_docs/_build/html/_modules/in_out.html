[docs]def input(model, reading):
    """
    This function imports the model and reading files into the correct mode

    Parameters
    ----------
    model : str
        Directory and filename of the file containing the model BioRECIPES spreadsheet
        Accepted files: .txt, .csv, .tsv, .xlsx
    reading: str
        Directory and filename of the files containing the machine reading spreadsheet output
        Accepted files: .txt, .csv, .tsv, .xlsx

    Returns
    -------
    new_model : pd.DataFrame
        Formatted model dataframe
        
    new_reading : pd.DataFrame]
        Formatted reading dataframe, including evidence count and list of PMCIDs

    graph : nx.Graph
        edgelist created from new_model
    """

    #Upload the model and reading files as dataframes based on the file extension
    model_ext = os.path.splitext(model)[1]
    reading_ext = os.path.splitext(reading)[1]

    if model_ext == '.txt': model_df = pd.read_csv(model, sep='\t',index_col=None).fillna("nan")
    elif model_ext == '.csv': model_df = pd.read_csv(model, sep=',',index_col=None).fillna("nan")
    elif model_ext == '.xlsx': model_df = pd.read_excel(model, index_col=None).fillna("nan")
    elif model_ext == '.tsv': model_df = pd.read_csv(model, sep='\t',index_col=None).fillna("nan")
    else: raise ValueError("The accepted file extensions are .txt, .csv, .xslx, and .tsv")

    if reading_ext == '.txt': reading_df = pd.read_csv(reading, sep='\t',index_col=None).fillna("nan")
    elif reading_ext == '.csv': reading_df = pd.read_csv(reading, sep=',',index_col=None).fillna("nan")
    elif reading_ext == '.xlsx': reading_df = pd.read_excel(reading,index_col=None).fillna("nan")
    elif reading_ext == '.tsv': reading_df = pd.read_csv(reading, sep='\t',index_col=None).fillna("nan")
    else: raise ValueError("The accepted file extensions are .txt, .csv, .xslx, and .tsv")

    #Accepted Column Names for reading and model spreadsheets
    reading_cols = ['Element Name','Element Type','Element Identifier','Location',
        'Location Identifier','Cell Line','Cell Type','Organism','PosReg Name',
        'PosReg Type','PosReg ID','PosReg Location','PosReg Location ID',
        'NegReg Name','NegReg Type','NegReg ID','NegReg Location','NegReg Location ID',
        'Interaction  Direct (D) or Indirect (I)','Paper ID','Evidence']
    model_cols = ['Element Name','Element Type','Element IDs','Cell line','Cell type',
        'Organism','Location','Location ID','Variable','Positive Regulators',
        'Negative Regulators']
    evidence_count_cols = ['Element Name','Element Type','Element Identifier','Location',
        'Location Identifier','Cell Line','Cell Type','Organism','PosReg Name',
        'PosReg Type','PosReg ID','PosReg Location','PosReg Location ID',
        'NegReg Name','NegReg Type','NegReg ID','NegReg Location','NegReg Location ID',
        'Interaction  Direct (D) or Indirect (I)']

    if {(set(model_cols).issubset(set(model_df.columns))) and 
        (set(reading_cols).issubset(set(reading_df.columns)))}: 
        #Format the dataframes
        new_reading = evidence_count(reading_df,evidence_count_cols)
        new_model = add_regulator_names_id(model_df)
        new_model = new_model.apply(lambda x: x.astype(str).str.lower())
        graph = node_edge_list(new_model)

        ## Check that dataframes have been created correctly
        # new_model.to_csv(r'MODEL_dataframe.csv')
        # new_reading.to_csv(r'READING_dataframe.csv')
    else: raise ValueError("reading column names should be: "+str(reading_cols)+"\n"+
       "model column names should be: "+ str(model_cols))
    return new_model, new_reading, graph



[docs]def output(reading_df,file_name):
    """
    This function outputs the scored reading interactions. 
    This writes output files, there are no return variables

    Parameters
    ----------
    reading_df : pd.DataFrame
        Dataframe of the scored reading dataframe
    file_name : str
        Directory and filename of the output suffix
    """

    #Output with all reading interactions, sorted by highest score    
    outputdf = reading_df.sort_values(by='Total Score',ascending=False)
    output_file = file_name+'outputDF.csv'
    outputdf.to_csv(output_file)

    ## Corroborations ##
    corr = reading_df[(reading_df['Kind Score'] == 1) | (reading_df['Kind Score'] == 2)]
    corr = corr.sort_values(by='Total Score',ascending=False)
    corr_file = file_name+'corroborations.csv'
    corr.to_csv(corr_file)

    ## Extensions ##
    ext = reading_df[(reading_df['Kind Score'] == 40) | (reading_df['Kind Score'] == 30)]
    ext = ext.sort_values(by='Total Score',ascending=False)
    ext_file = file_name+'extensions.csv'
    ext.to_csv(ext_file)

    ## Contradictions ##
    cont = reading_df[(reading_df['Kind Score'] == 10)]
    cont = cont.sort_values(by='Total Score',ascending=False)
    cont_file = file_name+'contradictions.csv'
    cont.to_csv(cont_file)

    # ## Special Cases ##
    que = reading_df[(reading_df['Kind Score'] == 20)]
    que = que.sort_values(by='Total Score',ascending=False)
    que_file = file_name+'questionable.csv'
    que.to_csv(que_file)

    return
