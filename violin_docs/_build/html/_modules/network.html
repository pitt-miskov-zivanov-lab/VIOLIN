[docs]def node_edge_list(model_df):
    """
    This function converts the model from a list of interactions into a node-edge list for use with NetworkX

    Parameters
    ----------
    model_df : pd.DataFrame
        The model dataframe

    Returns
    -------
    node_edge_list : nx.Graph
        An edgelist representation of the model
    """

    #Subset of the model, just element and regulator columns
    graph = model_df[['Variable','Positive Regulators','Negative Regulators']]
    #removes 'nan' placeholder
    graph = graph.replace('nan','')

    #remove excess punctuation from the regulator cells
    graph['Positive Regulators'] = graph['Positive Regulators'].str.replace('[','').str.replace(']','').str.replace('\'','')
    graph['Negative Regulators'] = graph['Negative Regulators'].str.replace('[','').str.replace(']','').str.replace('\'','')
    
    #combine regulators into one column, separated by '_' symbol
    graph['Regulators'] = graph['Positive Regulators']+'-'+graph['Negative Regulators']
    graph = graph.drop(columns=['Positive Regulators','Negative Regulators'])

    #Split each row by '-' character
    # This allows us to assign weights so that we know the "direction" (positive/negative) of the regulator
    graph = graph.set_index(['Variable']).stack().str.split('-', expand=True).stack().unstack(-2).reset_index(-1, drop=True).reset_index()
    
    #Assign 'weights' to edge defined edge, 0 for negative regulators, 1 for positive regulators
    for y in range(graph.shape[0]):
        if y%2 == 0: graph.at[y,'weight'] = 0
        else: graph.at[y,'weight'] = 1
    
    #Remove rows without a regulator node (housekeeping)
    graph = graph.replace(r'^\s*$', np.nan, regex=True).dropna()
    graph = graph.set_index(['Variable','weight']).stack().str.split(',', expand=True).stack().unstack(-2).reset_index(-1, drop=True).reset_index()
    # graph.to_csv(r'NodesEdges.csv')
    node_edge_list = nx.from_pandas_edgelist(graph,'Variable','Regulators','weight')
    
    return node_edge_list




[docs]def path_finding(reg,reg_dir,reg_type,reg_id,regd,regd_type,regd_id,model_df,graph):
    """
    This function searches for a path between the reading regulator and regulated in the model,
    and calculates the kind score based on the results

    Parameters
    ----------
    reg : str
        Element name of regulator node
    reg_dir : str
        Direction of regulator node (positive or negative)
    reg_type : str
        Type of regulator node (protein, protein family, rna, etc)
    reg_id : str
        ID number of regulator node
    regd : str
        Element name of the regulated node
    regd_type : str
        Type of regulated node
    regd_id : str
        ID number of regulated node
    model_df : pd.DataFrame
        Model dataframe
    graph : nx.Graph
        Model edgelist to create network for finding paths between elements

    Returns
    -------
    kind : int
        Kind Score value for the interaction
    """
    
    #Get regulator type (positive or negative)
    if reg_dir == 'NegReg Name': direction = 1
    else: direction = 0

    #Convert regulated to model variable name
    if find_element_index('name',regd,regd_type,model_df) != -1:
        regd_loc = find_element_index('name',regd,regd_type,model_df)
        regulated = model_df.at[regd_loc,'Variable']
    else:
        regd_loc = find_element_index('id',regd_id,regd_type,model_df)
        regulated = model_df.at[regd_loc,'Variable']

    #Convert regulator to model variable name
    if find_element_index('name',reg,reg_type,model_df) != -1:
        reg_loc = find_element_index('name',reg,reg_type,model_df)
        regulator = model_df.at[reg_loc,'Variable']
    else:
        reg_loc = find_element_index('id',reg_id,reg_type,model_df)
        regulator = model_df.at[reg_loc,'Variable']
    
    #Have to make sure regulator and regulated are in the directed path representation of the model
    #Some nodes may be in the model, but aren't regulated/regulators anywhere
    if regulator in graph and regulated in graph:
        #If there is a path between the regulator and regulated
        if nx.has_path(graph,regulator,regulated) and len(nx.shortest_path(graph,source=regulator,target=regulated))>1:
            #path list
            path = nx.shortest_path(graph,source=regulator,target=regulated,weight='weight')
            #first edge of the path
            start = graph[path[0]][path[1]]['weight']
            #Questionable - direct interaction, could indicated feed-forward loop
            if len(path) == 2: kind = 20
            #Contradiction - indirect interaction, starts with opposite regulation
            elif len(path) >2 and direction != start: kind = 10

            #check the remaining edges in the path for a contradiction
            else:
                idx = 1
                #Strong Corroboration - If no contradiction is found
                kind = 2
                #Put variable here to denote correct direction
                while idx < len(path)-1:
                    edge = graph[path[idx]][path[idx+1]]['weight']
                    if edge == start: idx += 1
                    else: 
                        #Contradiction - edge weight is opposite to reading interaction
                        kind = 10
                        break
        #Questionable - model interaction is self-regulation
        elif nx.has_path(graph,regulator,regulated) and len(nx.shortest_path(graph,source=regulator,target=regulated))==1: kind = 20
        #Extension - no path is between the 2 nodes 
        else: kind = 40
    #Extension - regulator or regulation node missing from directed graph
    else: kind = 40

    return kind
